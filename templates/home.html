import pandas as pd import matplotlib.pyplot as plt import seaborn as sns # Charger votre DataFrame (remplacez 'df' par le nom de votre DataFrame) # df = ... # Assurez-vous que la colonne date est de type datetime df['date'] = pd.to_datetime(df['date']) # Analyse temporelle : Nombre d'événements par utilisateur au fil du temps plt.figure(figsize=(12, 6)) df.groupby([df['date'].dt.date, 'user']).size().unstack().plot() plt.title('Nombre d\'événements par utilisateur au fil du temps') plt.xlabel('Date') plt.ylabel('Nombre d\'événements') plt.legend(title='Utilisateur', loc='upper right', bbox_to_anchor=(1.25, 1)) plt.show() # Analyse hebdomadaire plt.figure(figsize=(12, 6)) df.groupby([df['date'].dt.to_period('W'), 'user']).size().unstack().plot() plt.title('Nombre d\'événements par utilisateur par semaine') plt.xlabel('Semaine') plt.ylabel('Nombre d\'événements') plt.legend(title='Utilisateur', loc='upper right', bbox_to_anchor=(1.25, 1)) plt.show() # Analyse des jours de la semaine df['day_of_week'] = df['date'].dt.day_name() plt.figure(figsize=(8, 5)) sns.countplot(x='day_of_week', data=df, hue='user') plt.title('Nombre d\'événements par jour de la semaine et par utilisateur') plt.xlabel('Jour de la semaine') plt.ylabel('Nombre d\'événements') plt.xticks(rotation=45) plt.legend(title='Utilisateur', loc='upper right', bbox_to_anchor=(1.25, 1)) plt.show() # Analyse des heures de la journée df['hour_of_day'] = df['date'].dt.hour plt.figure(figsize=(10, 6)) sns.countplot(x='hour_of_day', data=df, hue='user') plt.title('Nombre d\'événements par heure de la journée et par utilisateur') plt.xlabel('Heure de la journée') plt.ylabel('Nombre d\'événements') plt.xticks(rotation=45) plt.legend(title='Utilisateur', loc='upper right', bbox_to_anchor=(1.25, 1)) plt.show() # Analyse de la durée entre les événements df['time_diff'] = df.groupby('user')['date'].diff().dt.total_seconds() / 3600 plt.figure(figsize=(10, 6)) sns.histplot(data=df, x='time_diff', bins=30, kde=True, hue='user', multiple='stack') plt.title('Distribution de la durée entre les événements par utilisateur') plt.xlabel('Durée entre les événements (heures)') plt.ylabel('Fréquence') plt.legend(title='Utilisateur', loc='upper right', bbox_to_anchor=(1.25, 1)) plt.show() #
